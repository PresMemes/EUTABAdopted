building_paradise_dome = {
	base_buildtime = @b2_time
	can_build = no

	category = amenity

	potential = {
		has_building = building_luxury_residence
		OR = {
			NOT = { is_planet_class = pc_habitat }
			AND = {
				has_planet_flag = advanced_habitat
				exists = owner
				owner = {
					has_ascension_perk = ap_voidborn
				}
			}
		}
	}

	allow = {
		has_upgraded_capital = yes
		owner = { is_ai = no }
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_gestalt = yes }
			owner = { has_valid_civic = civic_shared_burden }
			has_modifier = slave_colony
			owner = { is_ai = yes }
		}
	}

	convert_to = {
		building_organic_paradise
		building_drone_megastorage
		building_expanded_warren
		building_communal_housing_large
	}

	planet_modifier = {
		planet_housing_add = 6
		planet_amenities_add = 10
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_active_tradition = tr_domination_imperious_architecture
			}
		}
		modifier = {
			planet_housing_add = 1
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b2_minerals
			rare_crystals = @b2_rare_cost
		}
		upkeep = {
			energy = 3
			rare_crystals = @b2_rare_upkeep
		}
	}

	#eutab
	upgrades = {
		building_eutab_gaia_district # EUTAB
		building_avb_residence_3 # Additional Vanilla Buildings
	}

	prerequisites = {
		"tech_paradise_dome"
	}

	show_tech_unlock_if = {
		owner = {
			is_gestalt = no
			NOT = { has_valid_civic = civic_shared_burden }
		}
	}
}

building_maintenance_depot = {
	base_buildtime = 240
	can_build = no

	category = amenity

	potential = {
		owner = { is_gestalt = yes }
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_gestalt = no }
			AND = {
				owner = { is_ai = yes }
				free_amenities > 30
				free_building_slots = 0
			}
		}
	}

	planet_modifier = {
		job_maintenance_drone_add = 5
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = 2
		}
	}

	#eutab
	upgrades = {
		building_eutab_preservation_nexus # EUTAB
		building_avb_maintenance_depot_2 # Additional Vanilla Buildings
	}

	triggered_desc = {
		text = job_maintenance_drone_effect_desc
	}
}
