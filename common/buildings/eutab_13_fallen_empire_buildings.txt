building_nourishment_center = {
	base_buildtime = @b2_time
	can_build = yes
	can_demolish = yes
	#icon = building_food_processing_center #EUTAB REMOVE #PM: This basically forces the building to use building_nourishment_center.dds that EUTAB adds

	category = resources

	potential = {
		custom_tooltip = {
			exists = owner
			OR = {
				owner = { is_fallen_empire = yes }
				owner = {
					has_country_flag = nourishment_center_can_build
				}
				has_planet_flag = fallen_empire_world
			}
		}
	}

	allow = {
		custom_tooltip = {
			fail_text = ancrel.10004.failtext
			owner = {
				check_variable = { which = nourishment_center_build_count value >= 1 }
			}
		}
	}
	
	destroy_trigger = { always = no } 

	planet_modifier = {
	}

	resources = {
		category = planet_buildings
		produces = {
			food = 100
		}
		cost = {
			minerals = @b3_minerals
		}
		upkeep = {
			energy = 20
		}
	}

	on_queued = {
		owner = {
			subtract_variable = { which = nourishment_center_build_count value = 1 }
		}
	}

	on_unqueued = {
		owner = {
			change_variable = { which = nourishment_center_build_count value = 1 }
		}
	}

	upgrades = {
	}
}
